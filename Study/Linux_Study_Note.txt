3 开关机、重启、登陆注销和用户管理
	shutdown
		shutdown -h now	立即关机
		shutdown -h 1	一分钟之后关机
		shurdown -r now	立即重启
	halt
		关机
	reboot
		重启
	syn
		内存数据同步到磁盘
	su	
		登陆root用户。登陆系统时尽量别用root用户登陆，利用普通用户登陆然后用su命令登陆root用户, exit退出root用户。
	su - 用户名
		登陆指定用户（切换用户）
	logout	
		注销（在图形界面下无效，在级别3下有效）
	useradd 用户名
		添加用户
	useradd -d 指定目录 用户名
		在指定目录添加用户
	useradd -g 用户组 用户名
		在指定组添加用户
	usermod -g 用户组 用户名
		更改用户组
	passwd 用户名
		给指定用户添加或者修改密码
	userdel	用户名
		删除用户
	userdel -r 用户名
		删除用户及其家目录
	id 用户名
		查看用户信息
	groupadd 组名
		添加组
	groupdel 组名
		删除组
	etc/passwd
		用户配置文件
	etc/shadow
		用户密码（加过密）
	etc/group
		组配置文件
		Page.49

4 实用指令
	命令1 | 命令2
		命令1的结果用作命令2的参数
	命令1 || 命令2
		命令1执行失败再执行命令2
	命令 &
		将命令在后台执行
	命令1 && 命令2
		命令1执行成功后再执行命令2（类似与命令队列）
		
	运行级别
	0	关机
	1	单用户（用于root找回密码）
	2	多用户无网络
	3	多用户有网络
	4	系统未使用保留给用户
	5	图形界面
	6	系统重启
	常用的是3和5。
	init 数字选项
		改变当前运行级别
	man 命令或配置文件
		获取帮助信息
	help 命令
		获得shell内置命令的帮助信息
	pwd
		显示当前工作目录的绝对路径
	ls
		-a 显示当前目录中包括隐藏的所有文件和目录
		-l 以列表的方式显示信息
	cd 参数
		change directory改变路径
	cd /
		去根目录
	cd或者 cd ~
		去家目录
	mkdir 目录名
		创建目录（文件夹）
		-p 创建多级目录
		（如果要删除非空目录，用 rm -rf)
	touch 文件名
		创建文件
	cp 文件名 目录		
		复制文件到指定目录
		-r 递归复制整个文件夹
		\cp 强制覆盖
	rm 文件或目录
		删除
		-r 递归删除整个文件夹
		-f 强制执行不提示
	mv 文件或目录名 文件或目录名
		改名
	mv 文件名或目录 目录
		移动文件或目录
	cat 文件
		在terminal查看文件内容
		-n 显示行号
		| more 分页浏览
	more	
		基于vi编辑器的文本过滤器，全屏显示文本
		space	翻下一页
		Enter	下一行
		q	离开
		Ctrl+F	向下滚动一屏
		Ctrl+B	返回上一屏
		=	输出当前行号
		:f	输出当前文件名和行号
	less
		类似more，但比之更强大，支持各种显示终端，加载时不是一次加载整个文件而是根据需要加载内容，所以在加载大型文件时效率比较高
		space	下一页
		pagedown下一页
		pageup	上一页
		/字串	向下搜寻字串 n下一个 N上一个
		？字串	向上搜寻字串 n上一个 N下一个
	>
		输出重定向 向指定目标输出结果并覆盖
	>>
		追加 向指定目标输出结果，追加到文件尾部
	echo
		输出内容到控制台
	echo $PATH
		输出当前环境路径
	head 文件名
		查看文件前10行内容
		-n 5 文件名	查看前5行内容
	tail 文件名
		查看文件最后10行内容
		-n 5 文件名	查看最后5行内容
		-f 文件名	实时追踪文档的更新
	ln -s 原文件或目录


	wc 
		数字数  -l 行数
	^文件首
	$文件尾
